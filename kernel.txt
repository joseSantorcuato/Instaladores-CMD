sudo nano /etc/default/grub

GRUB_SAVEDEFAULT=true
GRUB_DISABLE_SUBMENU=y

sudo update-grub

sudo grep 'menuentry ' $GRUB_CONFIG | cut -f 2 -d "'" | nl -v 0

sudo grub-set-default 2

sudo grub-reboot 2
